## 思路一、常规思路（不推荐）
> 该思路是死磕for循环的一种，时间就不敢恭维了，我写的时候就知道它是绝对不会出现在实际应用中的

```js
var strStr = function(str, target) {
    if(!target){
        return 0
    }
    if(target.length > str.length){
        return -1
    }
    for(var i = 0; i< str.length; i++){
        let idx = i;
        if(str[i] == target[0]){
            for(let j = 0; j < target.length; j++){
                if(target[j] == str[idx]){
                    idx++
                }else{
                    break;
                }
            }
            if(idx - i == target.length){
                return i
            }else{
                continue;
            }
        }
    }
    return -1
};
```
