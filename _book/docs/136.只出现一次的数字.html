136.只出现一次的数字

只出现一次的数字

解法一（数学方法）

var singleNumber = function(nums) {
  var res = 0;
  if(nums.length < 2){
    return nums[0]
  }
  for(var i = 0 ;i < nums.length; i++ ){
    res ^= nums[i] //亦或  运用异或运算，任何一个数字异或它自己都等于0。
  }
  return res
};
